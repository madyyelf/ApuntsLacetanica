5. Playbooks

5.1. Playbook 1: Apt update && Upgrade + creació usuari admin

5.1.1. Creció usuari admin

- name: Crear usuari admin i afegir a sudo
  hosts: all
  become: true
  tasks:
    - name: Creacio usuari Admin
      ansible.builtin.user:
        name: admin
        password: "{{ 'admin123' | password_hash('sha512') }}"
        shell: /bin/zsh
        create_home: yes

    - name: Afegir a sudoers
      ansible.builtin.lineinfile:
        path: /etc/sudoers
        insertafter: '^root'
        line: 'admin ALL=(ALL:ALL) ALL'
        state: present

5.1.2. Actualització del sistema

- name: Actualitzacio
  hosts: all
  become: true
  tasks:
    - name: Update
      ansible.builtin.apt:
        update_cache: yes

    - name: Upgrade
      ansible.builtin.apt:
        upgrade: dist

5.3.1. Instal·lació LAMP

- import_playbook: update_upgrade.yml
- name: Instalar LAMP
  hosts: WEB
  become: true
  tasks:
    - name: Instalar LAMP
      ansible.builtin.apt:
        pkg:
        - apache2
        - php
        - libapache2-mod-php
        - php-mysql
        - mariadb-server
